print "=== LIST TEST ===\n"

print "create + len\n"
set xs (list create a b c)
print (format "$(list len $xs)\n")

print "get\n"
print (format "$(list get $xs 0)\n")
print (format "$(list get $xs 1)\n")
print (format "$(list get $xs 2)\n")

print "set\n"
list set $xs 1 z
print (format "$(list get $xs 1)\n")

print "push\n"
list push $xs d
print (format "$(list len $xs)\n")
print (format "$(list get $xs 3)\n")

print "pop\n"
set last (list pop $xs)
print (format "${last}\n")
print (format "$(list len $xs)\n")

print "append\n"
set ys (list create x y)
list append $xs $ys
print (format "$(list len $xs)\n")
print (format "$(list get $xs 3)\n")
print (format "$(list get $xs 4)\n")

print "index_of\n"
print (format "$(list index_of $xs a)\n")
print (format "$(list index_of $xs y)\n")
print (format "$(list index_of $xs q)\n")

print "clear\n"
list clear $xs
print (format "$(list len $xs)\n")

  print (format "${item}\n")
